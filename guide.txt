-- creation de module postgres (raha mbola tsisy)

    - atao anaty modules ...  base/postgres/main
    - mamorona .xml
    <?xml version="1.0" encoding="UTF-8"?>
<module xmlns="urn:jboss:module:1.5" name="org.postgresql">
    <resources>
        <resource-root path="postgresql-42.6.0.jar"/>
    </resources>
    <dependencies>
        <module name="javax.api"/>
        <module name="javax.transaction.api"/>
    </dependencies>
</module>


configuration de datasource
    - miditra ao am cli ./jboss-cli.sh --connect
    - mila enregistrena le base je pense : /subsystem=datasources/jdbc-driver=postgresql:add(driver-name=postgresql,driver-module-name=org.postgresql,driver-class-name=org.postgresql.Driver)

    - on insere ca : 
            /subsystem=datasources/data-source=BanqueC:add(\
jndi-name=java:jboss/datasources/BanqueC,\
driver-name=postgresql,\
connection-url=jdbc:postgresql://localhost:5432/centralisateur_db,\
user-name=postgres,\
password=postgres,\
min-pool-size=5,\
max-pool-size=20,\
pool-prefill=true,\
transaction-isolation=TRANSACTION_READ_COMMITTED,\
check-valid-connection-sql="SELECT 1")
    
    - on enable la datasource : /subsystem=datasources/data-source=BanqueCP:enable
    - on teste la connexion : /subsystem=datasources/data-source=BanqueCP:test-connection-in-pool



creation de persistence.xml
<?xml version="1.0" encoding="UTF-8"?>
<persistence xmlns="http://jakarta.ee/xml/ns/persistence"
             xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
             xsi:schemaLocation="http://jakarta.ee/xml/ns/persistence
             http://jakarta.ee/xml/ns/persistence/persistence_3_0.xsd"
             version="3.0">

    <persistence-unit name="CentralisateurPU" transaction-type="JTA">
        <!-- DataSource JNDI configurée dans WildFly -->
        <jta-data-source>java:jboss/datasources/BanqueCentralisateur</jta-data-source>
        
        <!-- Propriétés Hibernate -->
        <properties>
            <!-- Dialect PostgreSQL -->
            <property name="hibernate.dialect" value="org.hibernate.dialect.PostgreSQLDialect"/>
            
            <!-- Stratégie de génération du schéma -->
            <property name="hibernate.hbm2ddl.auto" value="update"/>
            
            <!-- Affichage des requêtes SQL -->
            <property name="hibernate.show_sql" value="true"/>
            <property name="hibernate.format_sql" value="true"/>
            
            <!-- Configuration des logs -->
            <property name="hibernate.use_sql_comments" value="true"/>
            
            <!-- Configuration du cache de second niveau (optionnel) -->
            <property name="hibernate.cache.use_second_level_cache" value="false"/>
            <property name="hibernate.cache.use_query_cache" value="false"/>
            
            <!-- Configuration de la validation -->
            <property name="hibernate.validator.apply_to_ddl" value="false"/>
            
            <!-- Configuration de la connexion -->
            <property name="hibernate.connection.characterEncoding" value="UTF-8"/>
            <property name="hibernate.connection.useUnicode" value="true"/>
        </properties>
    </persistence-unit>
</persistence>
